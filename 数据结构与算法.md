## 数组
- [双指针：盛最多水的容器](https://leetcode-cn.com/problems/container-with-most-water/)
```c++
class Solution {
public:
    int maxArea(vector<int>& height) {
        /**
        双指针：头尾指针
        具体：
            1. 计算当前头尾区域的容积: [较低一侧作为高度] * [right - left 作为宽度]
            2. 若当前容积大于记录的最大容积，则更新最大容积
            3. 移动较低一侧的索引
            4. 头尾指针相遇时，停止循环
        **/
        
        int res = 0;

        int i = 0, j = height.size() - 1;

        while (i < j){
            int curArea = min(height[i], height[j]) * (j - i);
            res = max(res, curArea);

            if (height[i] < height[j]){
                i++;
            }else{
                j--;
            }
        }
        return res;
    }
};
```
- [双指针-快慢指针：移动零](https://leetcode-cn.com/problems/move-zeroes/)
```c++
class Solution {
public:
    void moveZeroes(vector<int>& nums) {
		/**
         双指针：快慢指针
         1. 快指针指向的为非零元素时，交换快慢指针指向的元素
         2. 此时，慢指针指向的元素确定为非零元素，因此将慢指针往后移动一位
		**/
        int j = 0;

        for (int i = 0; i < nums.size(); i++) {
            if (nums[i] != 0) {
                swap(nums[i], nums[j]);
                j++;
            }
        }
        
    }
};
```
- [动态规划：爬楼梯](https://leetcode-cn.com/problems/climbing-stairs/)
```c++
class Solution {
public:
    int climbStairs(int n) {
        /**
        想爬上n级台阶， 有fn种方法
        但是想到第n阶，有两种策略：爬1阶 or 爬2阶 到第n阶
            因此：fn = f(n-1) + f(n-2)
        **/
        
        if (n < 3) return n;

        int f1 = 1, f2 = 2;
        int fn;
        for (int i = 3; i <= n; i++) {
            fn = f1 + f2;
            f1 = f2;
            f2 = fn;
        }
        return fn;

    }
};
```

- [三指针：三数之和](https://leetcode-cn.com/problems/3sum/)
```c++
class Solution {
public:
    vector<vector<int>> threeSum(vector<int>& nums) {
        /**
        暴力法的时间复杂度为 O(n^3)
		进阶解法：
        1. 排序
        2. 固定第一个元素 k
        3. 从固定元素之后的元素，使用头尾指针 i, j
            若和大于0，表明尾指针该前移 j--
            否则头指针后移 i++
        4. 考虑去重，当k元素与上一个元素相同时，继续移动 k++
            当i元素与上一个元素相同时，继续后移 i++
            当j元素与下一个元素相同时，继续前移 j--
        5. 将满足条件的组合加入到 res 中后，要移动 i, j 指针，同时要注意去重
        **/

        vector<vector<int>> res;
        sort(nums.begin(), nums.end());

        for (int k = 0; k < nums.size(); k++) 
        	// 排好序的数组，若第一个元素已经大于0，则不可能存在满足条件的组合
            if (nums[k] > 0) break;  
            if (k > 0 && nums[k] == nums[k-1]) continue;  // 去重

            int i = k + 1, j = nums.size() - 1;
            int sum;

            while (i < j) {
                sum = nums[k] + nums[i] + nums[j];
                if (sum == 0) {
                    res.push_back({nums[k], nums[i], nums[j]});
                    i++;
                    j--;
                    while (i < j && nums[i] == nums[i-1]) {
                        i++;
                    }
                    while (i < j && nums[j] == nums[j+1]){
                        j--;
                    }
                } else if (sum < 0) {
                    i++;
                    while (i < j && nums[i] == nums[i-1]) {
                        i++;
                    }
                } else if (sum > 0) {
                    j--;
                    while (i < j && nums[j] == nums[j+1]){
                        j--;
                    }
                }
            }
        }

        return res;
    }
};

```



## 链表
- [哑巴节点：反转链表](https://leetcode-cn.com/problems/reverse-linked-list/)
```c++
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */

/** 
   迭代法：
   猜测：由于链表会被改变，因此需要使用哑巴节点进行辅助
**/
class Solution {
public:
    ListNode* reverseList(ListNode* head) {

        if (head == NULL) return NULL;

        ListNode* pre = NULL;
        ListNode* cur = head;
        ListNode* nxt;

        while (cur) {
            nxt = cur->next;

            cur->next = pre;
            pre = cur;
            cur = nxt;
        }

        return pre;

    }
};

/**
	递归法
		1. terminator 终止条件！可对比迭代法进行思考
       	2. process current level 处理当前层的逻辑，对比迭代法 
		3. drill down 递归
		4. reverse state if need 如果需要，恢复变量状态
**/
class Solution {
public:
    ListNode* reverseList(ListNode* head) {
        return reverse(head, NULL);
    }

    ListNode* reverse(ListNode* cur, ListNode* pre) {
        if (cur == NULL) return pre;  // 当cur到链表尾时，pre为修改后的链表头节点

        ListNode* nxt = cur->next;
        cur->next = pre;
        pre = cur;
        cur = nxt;

        return reverse(cur, pre);
    }
};
```
- [哑巴节点：两两交换链表中的节点](https://leetcode-cn.com/problems/swap-nodes-in-pairs/)
```c++
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */

 /**
    迭代法：
        使用哑巴节点进行辅助，因为原链表被修改了
        链表的题目一定要细心
        比如：在修改该节点前，一定要先将该节点的next节点单独提取出来
 **/
class Solution {
public:
    ListNode* swapPairs(ListNode* head) {

        if (head == NULL || head->next == NULL) return head;

        ListNode* dummy = new ListNode();
        dummy->next = head;
        ListNode* pre = dummy;
        ListNode* cur = head;
        ListNode* nxt;
        ListNode* temp;

        while (cur != NULL && cur->next != NULL) {
            nxt = cur->next;
            temp = nxt->next;

            pre->next = nxt;
            nxt->next = cur;
            cur->next = temp;

            pre = cur;
            cur = temp;
        }

        return dummy->next;
    }
};

/**
    递归法： 完全可以对照迭代法写！ 
            迭代 & 递归的逻辑完全相同
            需要理解的是递归的四部曲如何操作
    递归四部曲：
        1. terminator
        2. process current level
        3. drill down
        4. reverse state if needed 
 **/
class Solution {
public:
    void swap(ListNode*cur, ListNode* pre) {
        if (cur == NULL || cur->next == NULL) return;

        ListNode* nxt = cur->next;
        ListNode* temp = nxt->next;

        pre->next = nxt;
        nxt->next = cur;
        cur->next = temp;

        pre = cur;
        cur = temp;

        swap(cur, pre);
    }

    ListNode* swapPairs(ListNode* head) {
        ListNode* dummy = new ListNode();
        dummy->next = head;
        ListNode* pre = dummy;
        swap(head, pre);
        return dummy->next;
    }
};
```
